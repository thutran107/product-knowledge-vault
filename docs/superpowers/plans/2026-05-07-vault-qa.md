> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Knowledge Q&A tab to Patrick App's dashboard that answers product questions by navigating the Anduin knowledge vault using a two-phase LLM retrieval loop, streaming answers word-by-word to the UI.

**Architecture:** Vault logic lives entirely in `src/lib/vault/` — a pure module with no Next.js dependencies — so it can be imported by a future agent layer without changes. The API route is a thin auth + stream wrapper. Phase 1 sends `index.md` + question to the LLM and gets back a JSON list of page paths. Pages are read in parallel from the filesystem. Phase 2 streams the synthesis back to the browser via Server-Sent Events. A confidence score derived from `source_count` frontmatter is returned with every answer.

**Tech Stack:** Next.js 16 App Router, TypeScript, Node.js `fs` (no new packages), existing `getLlmClient()` / `getDefaultModel()` from `src/lib/llm/client.ts`, Vitest, Server-Sent Events for streaming.

// src/lib/vault/reader.ts
import { readFileSync, existsSync } from 'fs';
import path from 'path';

export interface VaultEntry {
  path: string;
  title: string;
  type: string;
}

export interface VaultPage {
  path: string;
  title: string;
  type: string;
  sourceCount: number;
  content: string;
}

export interface ReadPageResult {
  page: VaultPage | null;
  error?: string;
  suggestions?: string[];
}

let _indexCache: VaultEntry[] | null = null;
let _indexCacheTime = 0;
const CACHE_TTL_MS = 5 * 60 * 1000;

export function clearCache(): void {
  _indexCache = null;
  _indexCacheTime = 0;
}

export function getVaultRoot(): string {
  const vaultPath = process.env.VAULT_PATH;
  if (!vaultPath) throw new Error('VAULT_PATH environment variable is not set');
  return vaultPath;
}

export function parseIndex(indexContent: string): VaultEntry[] {
  const entries: VaultEntry[] = [];
  const rowRegex = /\|\s*\[\[([^\]]+)\]\]\s*\|\s*([^|]+?)\s*\|/g;
  let match;
  while ((match = rowRegex.exec(indexContent)) !== null) {
    const entryPath = match[1].trim();
    const title = match[2].trim();
    if (!entryPath.startsWith('wiki/')) continue;
    const type = entryPath.split('/')[1] ?? 'page';
    entries.push({ path: entryPath, title, type });
  }
  return entries;
}

export function normalizePath(requested: string): string {
  return requested
    .replace(/\.md$/, '')
    .toLowerCase()
    .replace(/_/g, '-')
    .replace(/^(?!wiki\/)(.+)/, 'wiki/$1');
}

export function resolvePath(requested: string, knownPaths: string[]): string | null {
  const normalized = normalizePath(requested);
  if (knownPaths.includes(normalized)) return normalized;

  const slug = normalized.split('/').pop() ?? '';
  if (slug.length <= 3) return null;
  return knownPaths.find(p => p.includes(slug)) ?? null;
}

function parseSourceCount(content: string): number {
  const match = content.match(/source_count:\s*(\d+)/);
  return match ? parseInt(match[1], 10) : 0;
}

export function getIndexEntries(): VaultEntry[] {
  if (_indexCache && Date.now() - _indexCacheTime < CACHE_TTL_MS) {
    return _indexCache;
  }
  const root = getVaultRoot();
  const content = readFileSync(path.join(root, 'index.md'), 'utf-8');
  _indexCache = parseIndex(content);
  _indexCacheTime = Date.now();
  return _indexCache;
}

export function readIndex(): string {
  const root = getVaultRoot();
  return readFileSync(path.join(root, 'index.md'), 'utf-8');
}

export function readPage(requestedPath: string): ReadPageResult {
  const root = getVaultRoot();
  const entries = getIndexEntries();
  const knownPaths = entries.map(e => e.path);

  const resolved = resolvePath(requestedPath, knownPaths);

  if (!resolved) {
    const slug = normalizePath(requestedPath).split('/').pop() ?? '';
    const suggestions = knownPaths
      .filter(p => p.split('/').pop()?.slice(0, 4) === slug.slice(0, 4))
      .slice(0, 3);
    return { page: null, error: `Page not found: ${requestedPath}`, suggestions };
  }

  const filePath = path.join(root, resolved + '.md');
  if (!existsSync(filePath)) {
    return { page: null, error: `File missing for known path: ${resolved}` };
  }

  const content = readFileSync(filePath, 'utf-8');
  const entry = entries.find(e => e.path === resolved);
  return {
    page: {
      path: resolved,
      title: entry?.title ?? resolved,
      type: entry?.type ?? 'page',
      sourceCount: parseSourceCount(content),
      content,
    },
  };
}
```

// src/lib/vault/retrieval.ts
import { getLlmClient, getDefaultModel } from '@/lib/llm/client';
import { readIndex, readPage } from './reader';
import { NAVIGATION_PROMPT, SYNTHESIS_PROMPT } from './prompts';

export interface RetrievalResult {
  answer: string;
  sources: { path: string; title: string }[];
  confidence: 'high' | 'medium' | 'low';
}

export interface RetrievalStreamEvent {
  type: 'pages_selected' | 'chunk' | 'done' | 'error';
  pages?: { path: string; title: string }[];
  chunk?: string;
  result?: RetrievalResult;
  error?: string;
}

function computeConfidence(sourceCounts: number[]): 'high' | 'medium' | 'low' {
  const min = Math.min(...sourceCounts);
  if (min === 0) return 'low';
  if (min < 3) return 'medium';
  return 'high';
}

export async function selectPages(question: string): Promise<string[]> {
  const client = getLlmClient();
  const model = getDefaultModel();

  let response;
  try {
    response = await client.chat.completions.create({
      model,
      messages: [
        { role: 'system', content: NAVIGATION_PROMPT },
        { role: 'user', content: `Index:\n${readIndex()}\n\nQuestion: ${question}` },
      ],
      response_format: { type: 'json_object' },
      temperature: 0,
    });
  } catch {
    return [];
  }

  try {
    const content = response.choices[0]?.message?.content ?? '{}';
    const parsed = JSON.parse(content);
    const paths = Array.isArray(parsed.paths) ? parsed.paths : [];
    return paths.slice(0, 5);
  } catch {
    return [];
  }
}

export async function* runRetrievalStream(
  question: string
): AsyncGenerator<RetrievalStreamEvent> {
  // Phase 1: navigation
  const selectedPaths = await selectPages(question);
  const pageResults = selectedPaths.map(p => readPage(p));
  const pages = pageResults.filter(r => r.page !== null).map(r => r.page!);

  if (pages.length === 0) {
    yield {
      type: 'done',
      result: {
        answer: "The vault doesn't have enough information to answer this question yet.",
        sources: [],
        confidence: 'low',
      },
    };
    return;
  }

  yield {
    type: 'pages_selected',
    pages: pages.map(p => ({ path: p.path, title: p.title })),
  };

  // Phase 2: streaming synthesis
  const context = pages
    .map(p => `### ${p.title} (${p.path})\n\n${p.content}`)
    .join('\n\n---\n\n');

  const client = getLlmClient();
  const model = getDefaultModel();

  const stream = await client.chat.completions.create({
    model,
    messages: [
      { role: 'system', content: SYNTHESIS_PROMPT },
      { role: 'user', content: `Context:\n\n${context}\n\n---\n\nQuestion: ${question}` },
    ],
    stream: true,
    temperature: 0.2,
  });

  let fullAnswer = '';
  for await (const chunk of stream) {
    const delta = chunk.choices[0]?.delta?.content ?? '';
    if (delta) {
      fullAnswer += delta;
      yield { type: 'chunk', chunk: delta };
    }
  }

  yield {
    type: 'done',
    result: {
      answer: fullAnswer,
      sources: pages.map(p => ({ path: p.path, title: p.title })),
      confidence: computeConfidence(pages.map(p => p.sourceCount)),
    },
  };
}
```

// src/app/api/vault/ask/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { getServerSession } from '@/lib/auth/getServerSession';
import { runRetrievalStream } from '@/lib/vault/retrieval';

export const maxDuration = 60;

export async function POST(request: NextRequest) {
  const session = await getServerSession();
  if (!session?.user?.id) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }

  let question: string;
  try {
    const body = await request.json();
    question = (body.question ?? '').trim();
  } catch {
    return NextResponse.json({ error: 'Invalid request body' }, { status: 400 });
  }

  if (!question) {
    return NextResponse.json({ error: 'question is required' }, { status: 400 });
  }

  if (!process.env.VAULT_PATH) {
    return NextResponse.json({ error: 'Vault not configured on this server' }, { status: 503 });
  }

  const encoder = new TextEncoder();
  const stream = new ReadableStream({
    async start(controller) {
      try {
        for await (const event of runRetrievalStream(question)) {
          controller.enqueue(
            encoder.encode(`data: ${JSON.stringify(event)}\n\n`)
          );
        }
      } catch (err) {
        const message = err instanceof Error ? err.message : 'Unknown error';
        controller.enqueue(
          encoder.encode(`data: ${JSON.stringify({ type: 'error', error: message })}\n\n`)
        );
      } finally {
        controller.close();
      }
    },
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache',
      'Connection': 'keep-alive',
    },
  });
}
```

// src/app/dashboard/components/KnowledgeQATab.tsx
'use client';

import { useState } from 'react';
import { Send, BookOpen, Loader2, AlertCircle, FileText } from 'lucide-react';

interface Source {
  path: string;
  title: string;
}

type Confidence = 'high' | 'medium' | 'low';

interface QAState {
  pagesReading: Source[];
  answer: string;
  sources: Source[];
  confidence: Confidence | null;
  done: boolean;
}

const CONFIDENCE_CONFIG: Record<Confidence, { label: string; className: string }> = {
  high: { label: '', className: '' },
  medium: {
    label: 'Some pages have limited sources',
    className: 'bg-yellow-50 text-yellow-700 border border-yellow-200',
  },
  low: {
    label: 'Incomplete documentation — verify before using with customers',
    className: 'bg-red-50 text-red-700 border border-red-200',
  },
};

const EXAMPLE_QUESTIONS = [
  'How does DealCloud Order Creation work?',
  'What are the known limitations of Integration Hub?',
  'What is the difference between AES and QES signatures?',
];

export function KnowledgeQATab() {
  const [question, setQuestion] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [qa, setQA] = useState<QAState | null>(null);

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    const trimmed = question.trim();
    if (!trimmed || loading) return;

    setLoading(true);
    setError(null);
    setQA({ pagesReading: [], answer: '', sources: [], confidence: null, done: false });

    try {
      const res = await fetch('/api/vault/ask', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ question: trimmed }),
      });

      if (!res.ok) {
        const data = await res.json();
        throw new Error(data.error ?? 'Request failed');
      }

      const reader = res.body!.getReader();
      const decoder = new TextDecoder();

      while (true) {
        const { done, value } = await reader.read();
        if (done) break;

        const lines = decoder
          .decode(value)
          .split('\n')
          .filter(l => l.startsWith('data: '));

        for (const line of lines) {
          const event = JSON.parse(line.slice(6));

          if (event.type === 'pages_selected') {
            setQA(prev => prev ? { ...prev, pagesReading: event.pages } : prev);
          } else if (event.type === 'chunk') {
            setQA(prev => prev ? { ...prev, answer: prev.answer + event.chunk } : prev);
          } else if (event.type === 'done') {
            setQA(prev => prev
              ? { ...prev, sources: event.result.sources, confidence: event.result.confidence, done: true }
              : prev
            );
          } else if (event.type === 'error') {
            throw new Error(event.error);
          }
        }
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Something went wrong');
      setQA(null);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="max-w-3xl space-y-5">
      {/* Header */}
      <div>
        <h2 className="text-lg font-semibold text-slate-800">Product Knowledge Q&amp;A</h2>
        <p className="text-sm text-slate-500 mt-0.5">
          Ask any question about Anduin products, features, pricing, or implementation.
        </p>
      </div>

      {/* Example questions */}
      {!qa && !loading && (
        <div className="flex flex-wrap gap-2">
          {EXAMPLE_QUESTIONS.map(q => (
            <button
              key={q}
              onClick={() => setQuestion(q)}
              className="px-3 py-1.5 bg-slate-100 hover:bg-slate-200 text-slate-600 text-xs rounded-full transition-colors"
            >
              {q}
            </button>
          ))}
        </div>
      )}

      {/* Input */}
      <form onSubmit={handleSubmit} className="flex gap-2">
        <input
          type="text"
          value={question}
          onChange={e => setQuestion(e.target.value)}
          placeholder="Ask about any Anduin product, feature, or pricing..."
          className="flex-1 px-4 py-2.5 rounded-lg border border-slate-200 focus:outline-none focus:ring-2 focus:ring-blue-500 text-sm bg-white"
          disabled={loading}
        />
        <button
          type="submit"
          disabled={loading || !question.trim()}
          className="px-4 py-2.5 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50 disabled:cursor-not-allowed flex items-center gap-2 text-sm font-medium transition-colors"
        >
          {loading
            ? <Loader2 className="w-4 h-4 animate-spin" />
            : <Send className="w-4 h-4" />}
        </button>
      </form>

      {/* Error */}
      {error && (
        <div className="flex items-start gap-2 p-4 bg-red-50 border border-red-200 rounded-lg text-sm text-red-700">
          <AlertCircle className="w-4 h-4 mt-0.5 shrink-0" />
          {error}
        </div>
      )}

      {/* Response */}
      {qa && (
        <div className="space-y-3">
          {/* Pages being read */}
          {qa.pagesReading.length > 0 && (
            <div className="flex flex-wrap items-center gap-2">
              <span className="text-xs text-slate-400 flex items-center gap-1">
                <FileText className="w-3 h-3" />
                {qa.done ? 'Read:' : 'Reading:'}
              </span>
              {qa.pagesReading.map(p => (
                <span
                  key={p.path}
                  className="px-2 py-0.5 bg-slate-100 text-slate-500 text-xs rounded-full"
                >
                  {p.title}
                </span>
              ))}
            </div>
          )}

          {/* Streaming answer */}
          {qa.answer && (
            <div className="p-5 bg-white border border-slate-200 rounded-lg text-sm text-slate-800 leading-relaxed whitespace-pre-wrap">
              {qa.answer}
              {!qa.done && <span className="animate-pulse ml-0.5">▌</span>}
            </div>
          )}

          {/* Sources + confidence (after done) */}
          {qa.done && (
            <div className="space-y-2">
              {qa.sources.length > 0 && (
                <div className="flex flex-wrap items-center gap-2">
                  <span className="text-xs text-slate-400 flex items-center gap-1">
                    <BookOpen className="w-3 h-3" /> Sources:
                  </span>
                  {qa.sources.map(src => (
                    <span
                      key={src.path}
                      className="px-2 py-0.5 bg-slate-100 text-slate-600 text-xs rounded-full"
                    >
                      {src.title}
                    </span>
                  ))}
                </div>
              )}
              {qa.confidence && qa.confidence !== 'high' && (
                <p className={`text-xs px-3 py-1.5 rounded-lg ${CONFIDENCE_CONFIG[qa.confidence].className}`}>
                  ⚠ {CONFIDENCE_CONFIG[qa.confidence].label}
                </p>
              )}
            </div>
          )}
        </div>
      )}
    </div>
  );
}
```
